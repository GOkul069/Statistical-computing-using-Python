Plot 1: Top Product category by Average spend per unit¶

In [21]:
import pandas as pd
import matplotlib.pyplot as plt

tp = transactions.merge(products, on='product_id', how='left')
full = tp.merge(demographics, on='household_id', how='left')

full = full[full['quantity'] > 0]  
full['spend_per_unit'] = full['sales_value'] / full['quantity']

top10_cats = (full.groupby('product_category')['sales_value'].sum().nlargest(1).index)
df = full[full['product_category'].isin(top10_cats)]
agg = (df.groupby(['income', 'product_category'])['spend_per_unit'].mean().reset_index())

agg = agg.sort_values(['product_category', 'spend_per_unit'])

plt.figure(figsize=(14,6))

for cat in agg['product_category'].unique():
    sub = agg[agg['product_category'] == cat]
    plt.plot(sub['income'], sub['spend_per_unit'], marker='o', label=cat)

plt.title("Average Spend per Unit Across Income Groups\nTop Product Category")
plt.xlabel("Income Group")
plt.ylabel("Avg Spend per Unit (USD)")
plt.xticks(rotation=45)
plt.legend(title="Product Category", bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()

tp = transactions.merge(products, on="product_id", how="left")
full = tp.merge(demographics, on="household_id", how="left")

weekly = (full.groupby(["marital_status", "week", "household_id"])["sales_value"].sum().reset_index())

weekly_avg = (weekly.groupby(["marital_status", "week"])["sales_value"].mean().reset_index())

weekly_avg = weekly_avg[weekly_avg["marital_status"].isin(["Married", "Unmarried"])]

plt.figure(figsize=(12,6))
for grp in ["Married", "Unmarried"]:
    sub = weekly_avg[weekly_avg["marital_status"] == grp]
    plt.plot(sub["week"], sub["sales_value"], label=grp)

plt.title("Average Weekly Household Spending\nMarried vs Unmarried Shoppers")
plt.xlabel("Week of Year")
plt.ylabel("Average Weekly Spending")
plt.legend(title="Marital Status")
plt.tight_layout()
plt.show()

Plot 3: Average quantity purchased per trip¶

In [10]:
tp = transactions.merge(products, on="product_id", how="left")
full = tp.merge(demographics, on="household_id", how="left")

basket_qty = (full.groupby(["household_id", "basket_id", "department"])["quantity"].sum().reset_index())
basket_qty = basket_qty.merge(demographics[["household_id", "age"]],on="household_id",how="left")
top_departments = (basket_qty.groupby("department")["quantity"].sum().nlargest(8).index)

filtered = basket_qty[basket_qty["department"].isin(top_departments)]

avg_qty = (filtered.groupby(["age", "department"])["quantity"].mean().reset_index())

avg_qty = avg_qty.sort_values(["age", "quantity"])

plt.figure(figsize=(12,6))

for age_group in avg_qty["age"].unique():
    sub = avg_qty[avg_qty["age"] == age_group]
    plt.plot(sub["department"], sub["quantity"], marker="o", label=age_group)

plt.title("Average Quantity Purchased per Trip\nTop 8 Departments Across Age Groups")
plt.xlabel("Department")
plt.ylabel("Avg Units per Trip")
plt.xticks(rotation=45)
plt.legend(title="Age Group")
plt.tight_layout()
plt.show()
